# Store AWS certificate onto device (CA, CC, and PK)
# For each of the following commands, there will be a ">" response, after which you have to copy-paste the entire file contents and press ENTER, will respond with "OK"
AT+UDWNFILE="aws_ca.pem",1188
AT+UDWNFILE="383847e4d4-certificate.pem.crt",1224
AT+UDWNFILE="383847e4d4-private.pem.key",1679

# Verify availability of all modules on-device
# Each command will respond with "+ULSTFILE: <num> OK"
AT+ULSTFILE=2,"aws_ca.pem"
AT+ULSTFILE=2,"383847e4d4-certificate.pem.crt"
AT+ULSTFILE=2,"383847e4d4-private.pem.key"

# Import everything into the on-device Certificates Manager
# Each command will respond with "+USECMNG: <string> OK"
AT+USECMNG=1,0,"AWS_CA","aws_ca.pem"
AT+USECMNG=1,1,"AWS_Client","383847e4d4-certificate.pem.crt"
AT+USECMNG=1,2,"Client_Key","383847e4d4-private.pem.key"

# Configure the security profile
# Each command will respond with "OK"
AT+USECPRF=0
AT+USECPRF=0,0,1
AT+USECPRF=0,2,0
AT+USECPRF=0,3,"AWS_CA"
AT+USECPRF=0,5,"AWS_Client"
AT+USECPRF=0,6,"Client_Key"
AT+USECPRF=0,10,"xxx-ats.iot.us-east-2.amazonaws.com" (MIGHT NOT NEED THIS, DON'T USE IF NOT NECESSARY)

# AT DEVICE BOOT TIME
# Should always respond with something ending with "OK"
AT+UMQTT=2,"xxx-ats.iot.us-east2.amazonaws.com",8883				# Set remote server name (use correct endpoint here)
AT+UMQTT=11,1,0 													# Use secure connections
AT+UMQTTC=1 														# Connect to AWS broker and start secure MQTT session
AT+UMQTTC=4,0,"building/groundfloor/office_1/heating" 				# Subscribe to some topic
AT+UMQTTC=2,0,0,0,"building/groundfloor/office_1/temperature","10degrees Celsius" 		# Publish message to some topic
